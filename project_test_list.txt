User_service:

	User_controller:

		get "/" returns all users ✔️ ✔️

		post "/" creates user ✔️ ✔️

		put "/{user_id}" updates user✔️ ✔️

		delete "/{user_id}" deletes user ✔️ ✔️

		get "/is_registered/{user_id}" returns if a user is registered ✔️ ✔️

		get "/{user_id}" returns specific user by id✔️ ✔️

		put "/register_user/{user_id}" ✔️ ✔️

	Answer_controller:

		post "/" answers question ✔️

		put "/" updates answer ✔️

		delete "/user/{user_id}/question/{question_id}" deletes a specific answer ✔️

Poll client:

	Question_controller:

		get "/" returns all questions ✔️ ✔️

		post "/" creates question ✔️ ✔️

		get "/{question_id}" returns question by id ✔️ ✔️

		delete "/{question_id}" deletes question ✔️ ✔️

		get "/get_questions_by_user/{user_id}"  returns all questions that user answered ✔️ ✔️


	Answer_controller:

		post "/" answers question ✔️
	
		put "/" updates answer ✔️

		delete "/delete_user_answers/{user_id}" deletes all answers of user

		delete "/delete_answer/user/{user_id}/question/{question_id}" deletes answer ✔️

		get "get_user_answers/{user_id}"  returns all user answers ✔️

		get "get_user_answers_count/{user_id}" returns how many questions user answered ✔️

		get "get_all_questions_answers" returns all questions, their answers and how many users choose each answer ✔️

		get "check_user_answered/{user_id}/{question_id}" returns if user answered question ✔️

		get "get_question_answered_count/{question_id}" returns how many users answered ✔️

		get "get_questions_answers_count/{question_id}" returns how many users choose each of the questions ✔️
